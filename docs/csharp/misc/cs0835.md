## <a name="example"></a><span data-ttu-id="8fc61-108">Exemple</span><span class="sxs-lookup"><span data-stu-id="8fc61-108">Example</span></span>

using System.Linq.Expressions;

    public static int Main()

        Expression<int> e = x => x + 1; // CS0835

    }
