 <span data-ttu-id="f9015-107">L’exemple suivant génère l’erreur CS0061.</span><span class="sxs-lookup"><span data-stu-id="f9015-107">The following sample generates CS0061.</span></span>  
  
```csharp  
// CS0061.cs  
// compile with: /target:library  
internal interface A {}  
public interface AA : A {}  // CS0061  
  
// OK  
public interface B {}  
internal interface BB : B {}  
  
internal interface C {}  
internal interface CC : C {}  
```
