{  
   public class A { }  
}  
```  
  
 <span data-ttu-id="46cd3-109">Compilez ensuite ce fichier :</span><span class="sxs-lookup"><span data-stu-id="46cd3-109">Then, compile this file:</span></span>

using System;  
  
namespace Util   
{  
   public class A { }  
}  
  
public class Test   
{  
   public static void Main()   
   {  
      Console.WriteLine(typeof(Util.A)); // CS0435  
   }  
}  
```
