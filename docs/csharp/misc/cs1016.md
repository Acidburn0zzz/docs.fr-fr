 <span data-ttu-id="0edee-104">Les arguments d’attribut sans nom doivent figurer avant les arguments nommés.</span><span class="sxs-lookup"><span data-stu-id="0edee-104">Unnamed attribute arguments must appear before the named arguments.</span></span>  
  
## <a name="example"></a><span data-ttu-id="0edee-105">Exemple</span><span class="sxs-lookup"><span data-stu-id="0edee-105">Example</span></span>  
 <span data-ttu-id="0edee-106">L’exemple suivant génère l’erreur CS1016 :</span><span class="sxs-lookup"><span data-stu-id="0edee-106">The following sample generates CS1016:</span></span>  
  
```csharp  
// CS1016.cs  
using System;  
  
[AttributeUsage(AttributeTargets.Class)]  
public class HelpAttribute : Attribute  
{  
    public HelpAttribute(string url)   // url is a positional parameter  
    {  
        m_url = url;  
    }  
  
    public string Topic = null;      // Topic is a named parameter  
    private string m_url = null;  
}  
  
[HelpAttribute(Topic="Samples", "http://intranet/inhouse")]   // CS1016  
// try the following line instead  
//[HelpAttribute("http://intranet/inhouse", Topic="Samples")]  
public class MainClass  
{  
    public static void Main ()  
    {  
    }  
}  
```
