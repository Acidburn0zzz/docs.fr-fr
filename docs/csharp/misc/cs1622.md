 <span data-ttu-id="e245c-107">L’exemple suivant génère l’erreur CS1622 :</span><span class="sxs-lookup"><span data-stu-id="e245c-107">The following sample generates CS1622:</span></span>  
  
```csharp  
// CS1622.cs  
// compile with: /target:library  
using System.Collections;  
  
class C : IEnumerable  
{  
   public IEnumerator GetEnumerator()  
   {  
      return (IEnumerator) this;  // CS1622  
      yield return this;   // OK  
   }  
}  
```
