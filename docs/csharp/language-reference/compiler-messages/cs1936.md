## <a name="example"></a><span data-ttu-id="95c64-110">Exemple</span><span class="sxs-lookup"><span data-stu-id="95c64-110">Example</span></span>

{  
    static int Main()

        IEnumerable e = from x in obj // CS1936

        return 0;  
    }
